@Component({
  selector: 'app-root',
  imports: [RouterOutlet, WeatherWidgetComponent],
  templateUrl: './app.html',
  styleUrl: './app.css'
})
export class App {
  protected readonly title = signal('weather');

  weatherData: WeatherData = {
    location: 'The Hague',
    temperature: 13,
    condition: 'Partly Cloudy',
    humidity: 65,
    windSpeed: 12,
    icon: '⛅',
  };
}
